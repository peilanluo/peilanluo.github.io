// 简单冒泡排序，效率极差
func SimpleBubbleSort(arr []int) []int {
    for i := 0; i < len(arr); i++ {
        for j := i + 1; j < len(arr); j ++ {
            if arr[i] > arr[j] {
                arr[i], arr[j] = arr[j], arr[i]
            }
        } 
    }

    return arr
}

// 一般冒泡排序
func NormalBubbleSort(arr []int) []int {
    for i := 0; i < len(arr); i++ {
        for j := len(arr) - 1; j > i; j-- {
            if arr[j] < arr[j-1] {
                arr[j], arr[j-1] = arr[j-1], arr[j]
            }
        }
    }
}

// 优化冒泡排序
func OptimizingBubbleSort(arr []int) []int {
    status := true
    for i := 1; i < len(arr) && status; i++ {
        status = false
        for j := len(arr) - 1; j >= i; j-- {
            if arr[j] < arr[j-1] {
                arr[j], arr[j-1] = arr[j-1], arr[j]
                status = true
            }
        }
    }

    return arr
}

// 选择排序
func SelectionSort(arr []int) []int {
    for i := 0; i < len(arr); i++ {
        min := i
        for j := i + 1; j < len(arr); j++ {
            if arr[j] < arr[i] {
                min = j
            }
        }

        if min != i {
            arr[i], arr[min] = arr[min], arr[i]
        }
    }
}

// 插入排序
func InsertionSort(arr []int) []int {
    for i := 1; i < len(arr); i++ {
        for j := i; j > 0 && arr[j] < arr[j-1]; j-- {
            arr[j], arr[j-1] = arr[j-1], arr[j]
        }
    }

    return arr
}

// QuickSort 快速排序
func QuickSort(arr []int, start, end int) []int {
    if start < end {
        i, j := start, end
        key := arr[(start + end)/2]
        for i <= j {
            // 如果左侧数据比中间数小，则跳过
            for arr[i] < key {
                i++
            }

            // 如果右侧数据比中间数大，则跳过
            for arr[j] > key {
                j--
            }

            // 如果左边的key仍小于右边的key，说明左边的数比中间大，而右边数比左边小，需要交换
            if i <= j {
                arr[i], arr[j] = arr[j], arr[i]
                i++
                j--
            }
        }

        if start < j {
            QuickSort(arr, start, j)
        }

        if end > i {
            QuickSort(arr, i, end)
        }
    }

    return arr
}

// MergeSort 归并排序
func MergeSort(arr []int, start, end int) {
	if start >= end {
		return
	}

	mid := (start + end) / 2
	MergeSort(arr, start, mid)
	MergeSort(arr, mid+1, end)
	merge(arr, start, mid, end)
}

// merge ...
func merge(arr []int, start, mid, end int) {
	tmp := []int{}
	s1, s2 := start, mid+1
	for s1 <= mid && s2 <= end {
		if arr[s1] > arr[s2] {
			tmp = append(tmp, arr[s2])
			s2++
		} else {
			tmp = append(tmp, arr[s1])
			s1++
		}
	}

	if s1 <= mid {
		tmp = append(tmp, arr[s1:mid+1]...)
	}

	if s2 <= mid {
		tmp = append(tmp, arr[s2:end+1]...)
	}

	for key, val := range tmp {
		arr[start+key] = val
	}
}

func HeapSort(arr []int) {
    max := len(arr)
    for i := len(arr)/2-1; i >= 0; i-- {
        sink(arr, i, max)
    }

    for i := max-1; i >= 0; i-- {
        swap(arr, 0, i)
        sink(arr, 0, i)
    }
}

// sink 构建大顶堆
func sink(arr []int, i, max int) {
    for {
        j := 2*i+1
        if j >= max {
            break
        }

        if j+1 < max && arr[j+1] > arr[j] {
            j ++
        }

        if arr[i] >= arr[j] {
            break
        }

        swap(arr, i, j)
        i = j
    }
}

func swap(arr []int, i, j int) {
    arr[i], arr[j] = arr[j], arr[i]
}
